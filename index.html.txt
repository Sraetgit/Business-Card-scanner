    <script>
        const cardImageInput = document.getElementById('cardImage');
        const imagePreviewContainer = document.getElementById('imagePreviewContainer');
        const imagePreview = document.getElementById('imagePreview');
        const scanButton = document.getElementById('scanButton');
        const loader = document.getElementById('loader');
        const statusMessage = document.getElementById('statusMessage');

        // Form fields
        const clientNameInput = document.getElementById('clientName');
        const positionInput = document.getElementById('position');
        const companyNameInput = document.getElementById('companyName');
        const emailInput = document.getElementById('email');
        const websiteInput = document.getElementById('website');
        const socialMediaInput = document.getElementById('socialMedia');
        const countryInput = document.getElementById('country');

        const addToListButton = document.getElementById('addToListButton');
        const clearFormButton = document.getElementById('clearFormButton');
        const exportButton = document.getElementById('exportButton');
        const cardListBody = document.getElementById('cardList');
        const placeholderRow = document.getElementById('placeholderRow');

        let scannedCards = [];
        let editingIndex = null;
        let imageBase64 = null;
        
        // --- Gemini API Configuration ---
        const API_KEY = "AIzaSyAmmfDnob8DkN1s2VxDw0_JhD9rJrtk244"; // Leave empty, handled by the environment
        const API_URL = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${API_KEY}`;
        
        const cardSchema = {
            type: "OBJECT",
            properties: {
                clientName: { type: "STRING", description: "The full name of the person on the card." },
                position: { type: "STRING", description: "The job title or position of the person." },
                companyName: { type: "STRING", description: "The name of the company." },
                email: { type: "STRING", description: "The primary email address." },
                website: { type: "STRING", description: "The company's official website URL." },
                socialMedia: { type: "STRING", description: "Any social media handles or URLs (e.g., LinkedIn, Twitter), concatenated into a single string." },
                country: { type: "STRING", description: "The country where the company is located, inferred from the address." }
            }
        };


        // --- Event Listeners ---
        cardImageInput.addEventListener('change', (event) => {
            const file = event.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = (e) => {
                    imagePreview.src = e.target.result;
                    imageBase64 = e.target.result.split(',')[1];
                    imagePreviewContainer.classList.remove('hidden');
                    scanButton.disabled = false;
                };
                reader.readAsDataURL(file);
            }
        });

        scanButton.addEventListener('click', async () => {
            if (!imageBase64) {
                statusMessage.textContent = 'Please upload an image first.';
                return;
            }
            await extractInformation();
        });
        
        addToListButton.addEventListener('click', () => {
            const cardData = getFormData();
            if(!cardData.clientName && !cardData.companyName) {
                statusMessage.textContent = "Please provide at least a name or company name.";
                setTimeout(() => { if(statusMessage.textContent === "Please provide at least a name or company name.") statusMessage.textContent = ''; }, 3000);
                return;
            }
            statusMessage.textContent = '';

            if (editingIndex !== null) {
                scannedCards[editingIndex] = cardData;
            } else {
                scannedCards.push(cardData);
            }
            editingIndex = null;
            renderCardList();
            clearForm();
        });

        clearFormButton.addEventListener('click', clearForm);

        exportButton.addEventListener('click', exportToCSV);

        // --- Core Functions ---
        async function makeApiCall(payload) {
            try {
                const response = await fetch(API_URL, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                if (!response.ok) {
                    const errorBody = await response.text();
                    throw new Error(`API Error: ${response.status} ${response.statusText}\n${errorBody}`);
                }
                return await response.json();
            } catch (error) {
                console.error("API Call failed:", error);
                statusMessage.textContent = "An error occurred while contacting the AI service. Please try again.";
                throw error; // Re-throw to be caught by the caller
            }
        }

        function clearInputFields() {
            clientNameInput.value = '';
            positionInput.value = '';
            companyNameInput.value = '';
            emailInput.value = '';
            websiteInput.value = '';
            socialMediaInput.value = '';
            countryInput.value = '';
        }

        async function extractInformation() {
            setLoadingState(true, 'Analyzing image...');
            clearInputFields();
            
            try {
                // Step 1: Extract raw text from the image
                statusMessage.textContent = 'Step 1/2: Reading text from business card...';
                const ocrPayload = {
                    contents: [{
                        parts: [
                            { text: "Extract all text from this business card image. Transcribe it exactly as it appears." },
                            { inlineData: { mimeType: "image/jpeg", data: imageBase64 } }
                        ]
                    }]
                };
                const ocrResult = await makeApiCall(ocrPayload);
                const rawText = ocrResult.candidates?.[0]?.content?.parts?.[0]?.text;
                
                if (!rawText) {
                    throw new Error('Could not extract any text from the image.');
                }
                
                // Step 2: Structure the text into JSON
                statusMessage.textContent = 'Step 2/2: Structuring the extracted data...';
                const structuringPayload = {
                    contents: [{
                        parts: [{
                            text: `From the following text extracted from a business card, please populate the JSON schema. For any field where the information cannot be found, use the value "N/A".\n\n--- Business Card Text ---\n${rawText}\n--- End of Text ---`
                        }]
                    }],
                    generationConfig: {
                        responseMimeType: "application/json",
                        responseSchema: cardSchema
                    }
                };
                
                const structuredResult = await makeApiCall(structuringPayload);
                const jsonDataText = structuredResult.candidates?.[0]?.content?.parts?.[0]?.text;
                
                if (jsonDataText) {
                    const parsedData = JSON.parse(jsonDataText);
                    populateForm(parsedData);
                } else {
                     throw new Error('Failed to structure the extracted text.');
                }

            } catch (error) {
                console.error("Extraction failed:", error);
                statusMessage.textContent = `Error: ${error.message}`;
            } finally {
                setLoadingState(false);
            }
        }
        
        function populateForm(data) {
            clientNameInput.value = data.clientName || 'N/A';
            positionInput.value = data.position || 'N/A';
            companyNameInput.value = data.companyName || 'N/A';
            emailInput.value = data.email || 'N/A';
            websiteInput.value = data.website || 'N/A';
            socialMediaInput.value = data.socialMedia || 'N/A';
            countryInput.value = data.country || 'N/A';
            addToListButton.disabled = false;
        }

        function getFormData() {
            return {
                clientName: clientNameInput.value.trim(),
                position: positionInput.value.trim(),
                companyName: companyNameInput.value.trim(),
                email: emailInput.value.trim(),
                website: websiteInput.value.trim(),
                socialMedia: socialMediaInput.value.trim(),
                country: countryInput.value.trim(),
            };
        }

        function clearForm() {
            clearInputFields();
            
            cardImageInput.value = null;
            imagePreview.src = '#';
            imagePreviewContainer.classList.add('hidden');
            imageBase64 = null;
            editingIndex = null;

            addToListButton.textContent = 'Add to List';
            addToListButton.disabled = true;
            scanButton.disabled = true;
        }

        function renderCardList() {
            cardListBody.innerHTML = '';
            if (scannedCards.length === 0) {
                cardListBody.appendChild(placeholderRow);
                exportButton.disabled = true;
            } else {
                exportButton.disabled = false;
                scannedCards.forEach((card, index) => {
                    const row = document.createElement('tr');
                    row.className = "hover:bg-gray-50";
                    row.innerHTML = `
                        <td class="px-4 py-3 whitespace-nowrap text-sm font-medium text-gray-900">${card.clientName}</td>
                        <td class="px-4 py-3 whitespace-nowrap text-sm text-gray-500">${card.companyName}</td>
                        <td class="px-4 py-3 whitespace-nowrap text-sm text-gray-500">${card.email}</td>
                        <td class="px-4 py-3 whitespace-nowrap text-right text-sm font-medium">
                            <button onclick="editCard(${index})" class="text-indigo-600 hover:text-indigo-900 mr-3">Edit</button>
                            <button onclick="deleteCard(${index})" class="text-red-600 hover:text-red-900">Delete</button>
                        </td>
                    `;
                    cardListBody.appendChild(row);
                });
            }
        }
        
        window.editCard = (index) => {
            const card = scannedCards[index];
            populateForm(card);
            editingIndex = index;
            addToListButton.textContent = 'Update Entry';
            addToListButton.disabled = false;
        };

        window.deleteCard = (index) => {
            // Removed confirm() to avoid popups
            scannedCards.splice(index, 1);
            renderCardList();
        };

        function exportToCSV() {
            if (scannedCards.length === 0) {
                // Button is disabled, so this is just a safeguard. No alert needed.
                return;
            }

            const headers = [
                'Client Company Name', 
                'Company in which Country', 
                'Client Email address', 
                'Any social media references', 
                'Client official website', 
                'Position', 
                'Client Name'
            ];
            
            const rows = scannedCards.map(card => [
                card.companyName,
                card.country,
                card.email,
                card.socialMedia,
                card.website,
                card.position,
                card.clientName
            ]);

            let csvContent = "data:text/csv;charset=utf-t8," + headers.join(",") + "\n"
                + rows.map(e => e.map(cell => `"${cell.replace(/"/g, '""')}"`).join(",")).join("\n");
            
            const encodedUri = encodeURI(csvContent);
            const link = document.createElement("a");
            link.setAttribute("href", encodedUri);
            link.setAttribute("download", "business_cards.csv");
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }

        function setLoadingState(isLoading, message = '') {
            if (isLoading) {
                loader.classList.remove('hidden');
                scanButton.disabled = true;
                scanButton.classList.add('cursor-not-allowed');
            } else {
                loader.classList.add('hidden');
                scanButton.disabled = false;
                scanButton.classList.remove('cursor-not-allowed');
            }
            statusMessage.textContent = message;
        }

        // Initial state
        scanButton.disabled = true;
        addToListButton.disabled = true;
        exportButton.disabled = true;
    </script>
